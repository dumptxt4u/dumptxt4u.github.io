\end{center}

\begin{enumerate}

\item If 
\[
A=\begin{bmatrix}
\alpha & \beta\\
\gamma & -\alpha
\end{bmatrix}
\]
such that $A^2=I$, find the value of $1-\alpha^2-\beta\gamma$.

\item If 
\[
A=\begin{bmatrix}
0 & 1\\
-1 & 0
\end{bmatrix}
\]
and $(xI+yA)^2=A$, find the values of $x$ and $y$.

\item If 
\[
A=\begin{bmatrix}
a & b\\
c & d
\end{bmatrix}, \quad 
B=\begin{bmatrix}
p\\
q
\end{bmatrix}\neq
\begin{bmatrix}
0\\
0
\end{bmatrix}
\]
such that $AB=B$ and $a+d=5050$, find the value of $(ad-bc)$.

\item If 
\[
A=\begin{bmatrix}
1 & -2 & 3\\
-4 & 2 & 5
\end{bmatrix}, \quad
B=\begin{bmatrix}
2 & 3\\
4 & 5\\
2 & 1
\end{bmatrix}
\]
and $BA=[b_{ij}]$, find $b_{21}+b_{32}$.

\item Find the matrix $A^2$ where $A=[a_{ij}]$ is a $2\times2$ matrix whose elements are given by
\[
a_{ij}=\max(i,j)-\min(i,j).
\]

\item If $A$ and $B$ are square matrices of the same order and
\[
(A+B)^2=A^2+B^2,
\]
find $AB$.

\item If $A$ and $B$ are $3\times3$ matrices, then $A'B-B'A$ is always \underline{\hspace{3cm}}.

\item If $A$ is a $3\times4$ matrix and $B$ is a matrix such that both $A'B$ and $B'A$ are defined, find the order of matrix $B$.

\item If $(A+B)^T=
\begin{bmatrix}
2 & -1\\
7 & 6
\end{bmatrix}$
and 
\[
B=
\begin{bmatrix}
3 & -2\\
6 & 0
\end{bmatrix},
\]
find matrix $A$.

\item In an online advertisement campaign, there are two options for publicity: picture ads and video ads.  
Cost per ad is given by
\[
A=\begin{bmatrix}
0.2\\
0.5
\end{bmatrix}
\]
and the number of ads run by two companies $X$ and $Y$ is
\[
B=\begin{bmatrix}
100000 & 150000\\
8000 & 20000
\end{bmatrix}.
\]
Find the total cost of ads for both companies using matrix algebra.

\item If 
\[
A=\begin{bmatrix}
3 & a & -1\\
2 & 5 & c\\
b & 8 & 2
\end{bmatrix}
\]
is symmetric and
\[
B=\begin{bmatrix}
d & 3 & a\\
b-a & e & -2b-c\\
-2 & 6 & -f
\end{bmatrix}
\]
is skew-symmetric, find $AB$.

\item If 
\[
A=\begin{bmatrix}
1 & 2\\
3 & 4
\end{bmatrix}, \quad
B=\begin{bmatrix}
a & b\\
c & d
\end{bmatrix}
\]
and $AB=BA$, find
\[
\frac{a-d}{3b-c}.
\]

\item If 
\[
A=\begin{bmatrix}
1 & 1\\
0 & \beta
\end{bmatrix},
\]
for what value of $\beta$ is $A^2=I$?

\item State whether the following statement is true or false.  
If true, give a reason; if false, give a counterexample.  

\emph{If $A$ is a non-zero matrix such that $AK=O$, then $K$ must be a zero matrix.}

\item If 
\[
A=\begin{bmatrix}
2 & -3\\
1 & 0
\end{bmatrix}
\]
and $pA^2+qA+rI=O$, where $p,q,r$ are integers, find the value of $-\dfrac{q}{p}$.

\item Given 
\[
P=\begin{bmatrix}
3 & -1\\
1 & -2
\end{bmatrix},
\]
find a matrix $Q$ such that $P+Q$ is skew-symmetric.

\item If $\theta-\phi=\dfrac{\pi}{2}$ and
\[
A=\begin{bmatrix}
\cos^2\theta & \cos\theta\sin\theta\\
\cos\theta\sin\theta & \sin^2\theta
\end{bmatrix},\quad
B=\begin{bmatrix}
\cos^2\phi & \cos\phi\sin\phi\\
\cos\phi\sin\phi & \sin^2\phi
\end{bmatrix},
\]
find $AB$.

\item If $B$ is a symmetric matrix, prove that $BB^T$ is symmetric.

\item If $A$ is a square matrix of order $m$ and $(A^2-A)$ is invertible, prove that $A$ is invertible.

\end{enumerate}
